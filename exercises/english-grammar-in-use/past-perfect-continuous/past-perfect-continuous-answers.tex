    \item It \textbf{\textcolor{correct}{had been raining}} for hours.

\begin{notebox}[Spelling Notes]
\begin{itemize}

    \item \textbf{travel → traveling/travelling} (both spellings accepted)
\end{itemize}
\end{notebox}

\begin{notebox}[Key Concept: Cause and Visible Result]
The Past Perfect Continuous is often used to explain WHY something was visible or true at a past moment:
\begin{itemize}
    \item \textbf{Visible result:} He was out of breath. (past state)
    \item \textbf{Cause/Activity:} He had been running. (continuous activity before)
\end{itemize}
\end{notebox}

\section{Exercise 5: Past Perfect Continuous or Past Perfect Simple? (10 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{\textcolor{correct}{had been waiting}} — Focus on duration (continuous)
    \item \textbf{\textcolor{correct}{had read}} — Focus on completed quantity (simple)
    \item \textbf{\textcolor{correct}{had been swimming}} — Focus on activity causing visible result (continuous)
    \item \textbf{\textcolor{correct}{had finished}} — Focus on completion (simple)
    \item \textbf{\textcolor{correct}{had been arguing}} — Focus on duration (continuous)
    \item \textbf{\textcolor{correct}{had never seen}} — Experience, not process (simple)
    \item \textbf{\textcolor{correct}{had been working}} — Continuous activity causing tiredness (continuous)
    \item \textbf{\textcolor{correct}{had already eaten}} — Completed action (simple)
    \item \textbf{\textcolor{correct}{had been playing}} — Activity causing visible result (continuous)
    \item \textbf{\textcolor{correct}{had been studying}} — Prolonged activity (continuous)
\end{enumerate}
\end{answerbox}

\begin{notebox}[When to Use Each Tense]
\textbf{Past Perfect Continuous:}
\begin{itemize}
    \item Emphasizes \textbf{duration/process}: ``for hours'', ``all day'', ``since morning''
    \item Explains \textbf{visible results}: tired, wet, dirty, out of breath
\end{itemize}

\textbf{Past Perfect Simple:}
\begin{itemize}
    \item Emphasizes \textbf{completion}: finished, done, completed
    \item Focuses on \textbf{quantity}: read 3 books, eaten 5 apples
    \item Used with \textbf{stative verbs}: know, believe, want, like (NOT continuous)
\end{itemize}
\end{notebox}

\textbf{Scoring:} 1 point per correct answer. Total: 10 points.

    \item \textbf{\textcolor{correct}{A) had they been discussing}} — Duration question
    \item \textbf{\textcolor{correct}{B) had started}} — Completed event (NOT \textit{had been starting})
    \item \textbf{\textcolor{correct}{B) hadn't been sleeping}} — Continuous activity causing visible result
    \item \textbf{\textcolor{correct}{C) had been driving}} — Duration (``for three hours'')
    \item \textbf{\textcolor{correct}{B) had been raining}} — Continuous activity causing visible result
    \item \textbf{\textcolor{correct}{B) had been singing}} — Prolonged activity before another past event
    \item \textbf{\textcolor{correct}{A) had drunk}} — Focus on quantity (five cups)
    \item \textbf{\textcolor{correct}{B) had been thinking}} — Continuous process before that moment
\end{enumerate}
\end{answerbox}

\begin{notebox}[Common Trap]
\textbf{Some verbs are NOT normally used in continuous forms:}
\begin{itemize}
    \item \sout{had been starting} → \textbf{had started}
    \item \sout{had been knowing} → \textbf{had known}
    \item \sout{had been believing} → \textbf{had believed}
\end{itemize}
\end{notebox}

\section{Exercise 9: Error Correction (8 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{Error:} ``had been work'' → missing -ing\\
    \textbf{\textcolor{correct}{Correction:}} She \textbf{had been working} all day, so she was very tired.
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``have been waiting'' → wrong auxiliary (should be past)\\
    \textbf{\textcolor{correct}{Correction:}} They \textbf{had been waiting} for three hours when the train finally arrived.
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``had been knowing'' → ``know'' is a stative verb\\
    \textbf{\textcolor{correct}{Correction:}} I \textbf{had known} her for years before we started dating.
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``been wait'' → missing -ing\\
    \textbf{\textcolor{correct}{Correction:}} \textbf{Had you been waiting} long when I called?
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``since five years'' → ``since'' needs a point in time, not duration\\
    \textbf{\textcolor{correct}{Correction:}} He had been living in London \textbf{for five years} before he moved.
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``has been raining'' → wrong tense (should be past)\\
    \textbf{\textcolor{correct}{Correction:}} The ground was wet because it \textbf{had been raining}.
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``hadn't been believed'' → ``believe'' is a stative verb\\
    \textbf{\textcolor{correct}{Correction:}} She \textbf{hadn't believed} in herself until she won the competition.
    
    \vspace{0.2cm}
    
    \item \textbf{Error:} ``had she been studied'' → mixing forms\\
    \textbf{\textcolor{correct}{Correction:}} How long \textbf{had she been studying} before she passed the exam?
\end{enumerate}
\end{answerbox}

\textbf{Scoring:} 1 point per correct correction. Total: 8 points.
