\begin{grammarbox}[Grammar Summary: Past Perfect Continuous]
\textbf{Form:} \textit{had been} + \textbf{-ing} form of the verb

\vspace{0.3cm}

\begin{tabular}{ll}
\textbf{Affirmative:} & I/You/He/She/It/We/They \textbf{had been working} \\
\textbf{Negative:} & I/You/He/She/It/We/They \textbf{had not (hadn't) been working} \\
\textbf{Question:} & \textbf{Had} I/you/he/she/it/we/they \textbf{been working}? \\
\end{tabular}

\vspace{0.3cm}

\textbf{Main Uses:}
\begin{enumerate}[leftmargin=1.5cm]
    \item To talk about an activity that \textbf{started before a past moment and continued up to that moment}:\\
    \textit{``I had been waiting for an hour when the bus finally came.''}
    
    \item To explain \textbf{why something was happening} at a past moment (cause → visible result):\\
    \textit{``He was tired because he had been running.''}
    
    \item Often used with \textbf{for} (duration) and \textbf{since} (starting point):\\
    \textit{``They had been living there for 20 years before they moved.''}
\end{enumerate}
\end{grammarbox}

\vspace{0.3cm}

\begin{tipbox}[Tip: Past Perfect Continuous vs. Past Perfect Simple]
\begin{itemize}
    \item \textbf{Past Perfect Continuous} = emphasizes the \textbf{duration/process} of the activity\\
    \textit{``She had been reading for two hours.''} (focus on duration)
    
    \item \textbf{Past Perfect Simple} = emphasizes the \textbf{completion/result}\\
    \textit{``She had read five chapters.''} (focus on quantity completed)
\end{itemize}
\end{tipbox}

\end{itemize}

\section{Exercise 5: Past Perfect Continuous or Past Perfect Simple?}

\textit{Complete each sentence with the correct form: Past Perfect Continuous or Past Perfect Simple. Think about whether the focus is on duration/process or completion/result.}

\vspace{0.3cm}

\begin{enumerate}[label=\arabic*.]
    \item When I arrived, they \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (wait) for over an hour.\\
    \textit{(Focus: duration of waiting)}
    
    \item She \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (read) three books that week.\\
    \textit{(Focus: completed quantity)}
    
    \item His clothes were wet because he \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (swim).\\
    \textit{(Focus: activity in progress)}
    
    \item By 6 PM, I \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (finish) all my work.\\
    \textit{(Focus: completion)}
    
    \item They \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (argue) for hours before they finally agreed.\\
    \textit{(Focus: duration)}
    
    \item He \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (never / see) such a beautiful sunset.\\
    \textit{(Focus: experience, not process)}
    
    \item She was tired because she \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (work) since 6 AM.\\
    \textit{(Focus: continuous activity)}
    
    \item I \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (already / eat) when she offered me dinner.\\
    \textit{(Focus: completed action)}
    
    \item The children were covered in mud because they \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (play) in the garden.\\
    \textit{(Focus: visible result of activity)}
    
    \item Before the exam, he \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ (study) the same topic for weeks.\\
    \textit{(Focus: prolonged activity)}
\end{enumerate}

\vspace{0.3cm}

\begin{examtipbox}[{[TIP]} Cambridge Exam Tip — Key Word Transformations]
In Cambridge exams (FCE/CAE), you often need to transform sentences keeping the same meaning. Understanding when to use Past Perfect Continuous vs. Simple is crucial:
\begin{itemize}[leftmargin=1cm]
    \item \textbf{Duration/Process} → Past Perfect Continuous: \textit{``for hours'', ``all day'', ``since morning''}
    \item \textbf{Completion/Result} → Past Perfect Simple: \textit{``already'', ``just'', ``never'', ``three times''}
\end{itemize}
\end{examtipbox}

\begin{examtipbox}[{[MEMO]} Remember for Exams]
\textbf{Stative verbs} that should NOT be used in continuous form:
\begin{multicols}{3}
\begin{itemize}[leftmargin=0.5cm]
    \item know
    \item believe
    \item want
    \item like/love/hate
    \item understand
    \item remember
    \item belong
    \item own
    \item seem
\end{itemize}
\end{multicols}
\textbf{Exception:} Some verbs change meaning in continuous → \textit{``I had been thinking about it''} (process) is correct!
\end{examtipbox}

{\Large Past Perfect Continuous}\\[0.2cm]
\rule{0.8\textwidth}{0.4pt}
\end{center}

\vspace{0.3cm}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!70!black, title=\textbf{FORM}]
\begin{center}
\textbf{had} + \textbf{been} + \textbf{verb-ing}
\end{center}

\begin{tabular}{ll}
\textbf{(+)} & I/You/He/She/It/We/They \textbf{had been working} \\
\textbf{(-)} & I/You/He/She/It/We/They \textbf{hadn't been working} \\
\textbf{(?)} & \textbf{Had} I/you/he/she/it/we/they \textbf{been working}? \\
\end{tabular}
\end{tcolorbox}

\vspace{0.3cm}

\begin{tcolorbox}[colback=green!5!white, colframe=easygreen, title=\textbf{MAIN USES}]
\begin{enumerate}[leftmargin=1cm]
    \item \textbf{Duration before a past moment:}\\
    \textit{``I had been waiting for an hour when the bus came.''}
    
    \item \textbf{Cause → Visible result:}\\
    \textit{``He was tired because he had been running.''}
    
    \item \textbf{Repeated actions before a past event:}\\
    \textit{``She had been applying to jobs for months before she got hired.''}
\end{enumerate}
\end{tcolorbox}

\vspace{0.3cm}

\begin{tcolorbox}[colback=yellow!5!white, colframe=mediumyellow, title=\textbf{TIME EXPRESSIONS}]
\begin{tabular}{|l|l|}
\hline
\textbf{FOR} + duration & for hours, for weeks, for ages, for a long time \\
\hline
\textbf{SINCE} + starting point & since morning, since 2020, since January \\
\hline
\textbf{ALL} + time period & all day, all night, all morning, all week \\
\hline
\textbf{HOW LONG} questions & How long had you been waiting...? \\
\hline
\end{tabular}
\end{tcolorbox}

\vspace{0.3cm}

\begin{tcolorbox}[colback=red!5!white, colframe=hardred, title=\textbf{{[!]} COMMON MISTAKES}]
\begin{tabular}{ll}
{[X]} \textit{had been work} & {[OK]} \textit{had been \textbf{working}} \\
{[X]} \textit{had been knowing} & {[OK]} \textit{had \textbf{known}} (stative verb!) \\
{[X]} \textit{since 3 hours} & {[OK]} \textit{\textbf{for} 3 hours} \\
{[X]} \textit{have been waiting (past)} & {[OK]} \textit{\textbf{had} been waiting} \\
\end{tabular}
\end{tcolorbox}

\vspace{0.3cm}

\begin{tcolorbox}[colback=purple!5!white, colframe=toeflpurple, title=\textbf{PPC vs. PP Simple}]
\begin{tabular}{|p{7cm}|p{7cm}|}
\hline
\textbf{Past Perfect Continuous} & \textbf{Past Perfect Simple} \\
\hline
Focus on \textbf{duration/process} & Focus on \textbf{completion/result} \\
\textit{``She had been reading for hours.''} & \textit{``She had read 3 books.''} \\
\hline
\end{tabular}
\end{tcolorbox}
