\begin{grammarbox}[Grammar Summary: HAVE and HAVE GOT]
\textbf{1. For possession, relationships, illnesses, appointments:}

\vspace{0.2cm}

\begin{tabular}{ll}
\textbf{have} & = \textbf{have got} (same meaning) \\
\textit{I have a car.} & = \textit{I've got a car.} \\
\textit{She has two brothers.} & = \textit{She's got two brothers.} \\
\textit{He has a headache.} & = \textit{He's got a headache.} \\
\end{tabular}

\vspace{0.3cm}

\textbf{2. Questions and negatives — three forms:}

\vspace{0.2cm}

\begin{tabular}{lll}
\textit{Do you have...?} & \textit{Have you got...?} & \textit{Have you...?} (less usual) \\
\textit{I don't have...} & \textit{I haven't got...} & \textit{I haven't...} (less usual) \\
\end{tabular}

\vspace{0.3cm}

\textbf{3. For past tense:} Use \textbf{had} (NOT ``had got'')\\
\textit{Lisa had long hair when she was a child.}

\vspace{0.3cm}

\textbf{4. For actions/experiences} (have breakfast, have a shower, have fun, etc.):\\
Use \textbf{have} only (NOT ``have got''). Questions/negatives use \textbf{do/does/did}.\\
\textit{Did you have a good time?} (NOT \textit{Had you got a good time?})
\end{grammarbox}

\vspace{0.3cm}

\begin{tipbox}[Key Distinction: Auxiliary Verbs]
\textbf{When do you use DO/DOES/DID vs. HAVE/HAS/HAD?}
\begin{itemize}
    \item \textbf{Simple past questions/negatives} about states at a specific time → \textbf{did/didn't}\\
    \textit{Did you have a car when you lived in Paris?}
    
    \item \textbf{Past perfect} (earlier state before another past action) → \textbf{had/hadn't}\\
    \textit{I couldn't call you because I hadn't got my phone.}
    
    \item \textbf{Present habits/routines} → \textbf{do/does}\\
    \textit{How often do you have a shower?}
    
    \item \textbf{Experiences up to now} → \textbf{have/has}\\
    \textit{Have you ever had problems with your car?}
\end{itemize}
\end{tipbox}

\begin{pitfallbox}[Common Mistake for Brazilians]
In Portuguese, we often use ``ter'' for everything. But in English:
\begin{itemize}
    \item \textbf{Routines/habits} (how often, usually) → \textbf{DO/DOES} + have\\
    \textit{How often \textbf{do} you have lunch at home?} (NOT \textit{How often have you...})
    
    \item \textbf{Possession NOW} → \textbf{HAVE/HAS} got OR \textbf{DO/DOES} + have\\
    \textit{\textbf{Have} you got a car?} = \textit{\textbf{Do} you have a car?}
\end{itemize}
\end{pitfallbox}

    \item By noon yesterday, I \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ had time to eat — but later I did.
    
    \item \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ she have a good time at the party?
    
    \item He was hungry because he \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ had breakfast.
    
    \item We \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ have any trouble finding the restaurant.
    
    \item I couldn't pay because I \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ brought my wallet.
    
    \item \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ they have enough money to buy the house?
    
    \item The meeting was cancelled because she \_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_\_ had time to prepare.
\end{enumerate}

\vspace{0.3cm}

\begin{tipbox}[Understanding the Distinction]
\textbf{Simple Past (DID/DIDN'T):} Single past timeframe, no sequence needed.\\
\textit{I \textbf{didn't have} time yesterday.} (fact at a specific past time)

\textbf{Past Perfect (HAD/HADN'T):} Earlier state BEFORE another past action.\\
\textit{I couldn't call because I \textbf{hadn't got} my phone.} (not having the phone → couldn't call)

\textbf{Key question:} Is there a CAUSE before a RESULT? → Use past perfect.
\end{tipbox}

\begin{center}
{\Huge\bfseries Quick Reference}\\[0.3cm]
{\Large HAVE and HAVE GOT}\\[0.2cm]
\rule{0.8\textwidth}{0.4pt}
\end{center}

\vspace{0.5cm}

\begin{grammarbox}[When to Use Each Form]

\textbf{1. HAVE = HAVE GOT (for possession, relationships, illnesses)}
\begin{itemize}
    \item I \textbf{have} a car. = I'\textbf{ve got} a car.
    \item She \textbf{has} a headache. = She'\textbf{s got} a headache.
\end{itemize}

\vspace{0.3cm}

\textbf{2. Questions and Negatives}
\begin{center}
\begin{tabular}{|l|l|}
\hline
\textbf{With DO/DOES} & \textbf{With HAVE/HAS} \\
\hline
Do you have a car? & Have you got a car? \\
I don't have time. & I haven't got time. \\
Does she have a sister? & Has she got a sister? \\
\hline
\end{tabular}
\end{center}

\vspace{0.3cm}

\textbf{3. Past Tense — Use HAD (not ``had got'')}
\begin{itemize}
    \item I \textbf{had} a bike when I was young. (NOT \textit{I had got})
    \item \textbf{Did} you \textbf{have} a car? (NOT \textit{Had you got})
\end{itemize}

\vspace{0.3cm}

\textbf{4. HAVE for Actions — Use DO/DOES/DID}
\begin{itemize}
    \item How often \textbf{do} you \textbf{have} a shower? (NOT \textit{How often have you...})
    \item \textbf{Did} you \textbf{have} a good time? (NOT \textit{Had you got...})
    \item We'\textbf{re having} a party. (continuous OK for actions)
\end{itemize}

\end{grammarbox}

\vspace{0.5cm}

\begin{pitfallbox}[Auxiliary Verb Decision Tree]

\textbf{Asking about the PAST?}
\begin{itemize}
    \item Single past fact/state → \textbf{DID} + have\\
    \textit{Did you have any problems yesterday?}
    
    \item Cause BEFORE result (two past points) → \textbf{HAD} + had/got\\
    \textit{I couldn't pay because I hadn't got my wallet.}
\end{itemize}

\vspace{0.3cm}

\textbf{Asking about the PRESENT?}
\begin{itemize}
    \item Routine/habit (how often, usually) → \textbf{DO/DOES} + have\\
    \textit{How often do you have meetings?}
    
    \item Possession NOW → \textbf{HAVE/HAS} got OR \textbf{DO/DOES} + have\\
    \textit{Have you got a pen?} = \textit{Do you have a pen?}
    
    \item Experience up to now → \textbf{HAVE/HAS} + had\\
    \textit{Have you ever had sushi?}
\end{itemize}

\end{pitfallbox}
