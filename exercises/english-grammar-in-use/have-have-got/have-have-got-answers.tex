\section{Exercise 1: HAVE or HAVE GOT — Rewrite}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item She has three children. → \textbf{\textcolor{correct}{She's got three children.}}
    
    \item We've got a big garden. → \textbf{\textcolor{correct}{We have a big garden.}}
    
    \item He's got a headache. → \textbf{\textcolor{correct}{He has a headache.}}
    
    \item I have an appointment tomorrow. → \textbf{\textcolor{correct}{I've got an appointment tomorrow.}}
    
    \item They've got two dogs and a cat. → \textbf{\textcolor{correct}{They have two dogs and a cat.}}
    
    \item Our house has a small garden. → \textbf{\textcolor{correct}{Our house has got a small garden.}}
    
    \item Have you got any questions? → \textbf{\textcolor{correct}{Do you have any questions?}}
    
    \item Does she have a car? → \textbf{\textcolor{correct}{Has she got a car?}}
    
    \item I haven't got any money. → \textbf{\textcolor{correct}{I don't have any money.}}
    
    \item We don't have enough time. → \textbf{\textcolor{correct}{We haven't got enough time.}}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{notebox}[Key Point]
For possession, relationships, and states, \textbf{have} and \textbf{have got} are interchangeable in meaning. The difference is:
\begin{itemize}
    \item \textbf{Have} uses DO/DOES in questions and negatives
    \item \textbf{Have got} uses HAVE/HAS in questions and negatives
\end{itemize}
Both forms are equally correct in British and American English.
\end{notebox}

\section{Exercise 2: Make Questions — DO or HAVE?}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item Yes, she has two sisters.\\
    → \textbf{\textcolor{correct}{Does she have two sisters?}} OR \textbf{\textcolor{correct}{Has she got two sisters?}}
    
    \item No, I don't have any pets.\\
    → \textbf{\textcolor{correct}{Do you have any pets?}} OR \textbf{\textcolor{correct}{Have you got any pets?}}
    
    \item Yes, they've got a swimming pool.\\
    → \textbf{\textcolor{correct}{Have they got a swimming pool?}} OR \textbf{\textcolor{correct}{Do they have a swimming pool?}}
    
    \item No, he hasn't got a driving license.\\
    → \textbf{\textcolor{correct}{Has he got a driving license?}} OR \textbf{\textcolor{correct}{Does he have a driving license?}}
    
    \item Yes, our flat has a balcony.\\
    → \textbf{\textcolor{correct}{Does your flat have a balcony?}} OR \textbf{\textcolor{correct}{Has your flat got a balcony?}}
    
    \item No, I haven't got my passport with me.\\
    → \textbf{\textcolor{correct}{Have you got your passport with you?}} OR \textbf{\textcolor{correct}{Do you have your passport with you?}}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{tipbox}[Tip for Learners]
When you see \textbf{haven't/hasn't} in the answer → question uses \textbf{Have/Has ... got?}\\
When you see \textbf{don't/doesn't} in the answer → question uses \textbf{Do/Does ... have?}\\
But remember: both question forms are correct!
\end{tipbox}

\section{Exercise 3: HAVE for Actions — Complete the Sentences}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item What time do you usually \textbf{\textcolor{correct}{have}} breakfast?
    
    \item We \textbf{\textcolor{correct}{had}} a great time at the party last night.
    
    \item \textbf{\textcolor{correct}{Did}} you \textbf{\textcolor{correct}{have}} a good holiday?
    
    \item I'm going to \textbf{\textcolor{correct}{have}} a shower before dinner.
    
    \item She \textbf{\textcolor{correct}{has}} lunch with her colleagues every day.
    
    \item They didn't \textbf{\textcolor{correct}{have}} any problems finding the hotel.
    
    \item We're \textbf{\textcolor{correct}{having}} a party next Saturday. Would you like to come?
    
    \item \textbf{\textcolor{correct}{Do}} you usually \textbf{\textcolor{correct}{have}} coffee in the morning?
    
    \item I \textbf{\textcolor{correct}{had}} a dream last night about flying.
    
    \item Let's \textbf{\textcolor{correct}{have}} a break. I'm tired.
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{pitfallbox}[Critical Distinction]
When HAVE means an \textbf{action/experience} (have breakfast, have a shower, have fun):
\begin{itemize}
    \item NEVER use ``have got'' — \textcolor{red}{\st{I've got breakfast}} \checkmark \textit{I have breakfast}
    \item Questions use DO/DOES/DID — \textcolor{red}{\st{Have you a good time?}} \checkmark \textit{Did you have a good time?}
    \item CAN use continuous forms — \checkmark \textit{We're having a wonderful time!}
\end{itemize}
\end{pitfallbox}

\section{Exercise 4: Past Tense — HAD}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item When I was young, I \textbf{\textcolor{correct}{had}} long hair.
    
    \item She \textbf{\textcolor{correct}{had}} a beautiful dress at her wedding.
    
    \item Did you \textbf{\textcolor{correct}{have}} a car when you were living in Paris?
    
    \item They \textbf{\textcolor{correct}{had}} a big house before they moved to an apartment.
    
    \item I didn't \textbf{\textcolor{correct}{have}} time to go shopping yesterday.
    
    \item \textbf{\textcolor{correct}{Did}} Lisa \textbf{\textcolor{correct}{have}} any brothers or sisters?
    
    \item We \textbf{\textcolor{correct}{had}} a wonderful holiday in Italy last summer.
    
    \item He \textbf{\textcolor{correct}{had}} a really bad cold last week.
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{notebox}[Past Tense Rule]
For past possession/states, use \textbf{had} — NOT ``had got.''\\
Questions and negatives use \textbf{DID}: \textit{Did you have...?} / \textit{I didn't have...}\\
``Had got'' only appears in \textbf{past perfect} (earlier past before another past).
\end{notebox}

\vspace{0.3cm}

\begin{tipbox}[Why We Avoid ``Had Got'' — A Detailed Look]
\textbf{The reason we don't say ``I had got'' very often is because ``got'' and ``have got'' are already acting as substitutes for other words.}

When we move into the past tense, the language naturally simplifies. Here's the breakdown:

\vspace{0.2cm}
\textbf{1. ``Have got'' is a Present Tense ``Cheat''}

In British English, we use \textbf{``have got''} to mean \textbf{``possess''} right now.
\begin{itemize}
    \item \textbf{Present:} ``I \textbf{have} a car'' = ``I \textbf{have got} a car.''
    \item \textbf{Logic:} Even though ``got'' looks like past participle, ``have got'' functions as \textbf{present tense}.
\end{itemize}

\vspace{0.2cm}
\textbf{2. Past Tense is Simpler}

We already have a perfectly good, short word: \textbf{had}. Adding ``got'' feels redundant.
\begin{itemize}
    \item \textbf{Present:} ``I've got a new job.'' (Natural)
    \item \textbf{Past:} ``I \textbf{had} a new job.'' (Standard)
    \item \textbf{Past:} ``I \textbf{had got} a new job.'' (Clunky, unnecessary)
\end{itemize}

\vspace{0.2cm}
\textbf{3. Confusion with ``Obtaining''}

``Get'' has two meanings: \textbf{possess} and \textbf{obtain/become}. We save ``had got'' for obtaining actions.

\begin{center}
\begin{tabular}{|l|l|l|}
\hline
\textbf{Meaning} & \textbf{Present} & \textbf{Past} \\
\hline
Possession & I've got a cold & I \textbf{had} a cold \\
Obtaining & I get coffee daily & I \textbf{had got} coffee before the meeting \\
\hline
\end{tabular}
\end{center}

\vspace{0.2cm}
\textbf{4. Negatives and Questions}

``Hadn't got'' sounds old-fashioned to modern ears:
\begin{itemize}
    \item \textbf{Natural:} ``I \textbf{didn't have} any money.''
    \item \textbf{Rare/Clunky:} ``I \textbf{hadn't got} any money.''
\end{itemize}

\vspace{0.2cm}
\textbf{Summary:} If talking about \textbf{owning} something in the past → use \textbf{had}. If talking about the \textbf{action of receiving} before another past event → use \textbf{had got/gotten}.
\end{tipbox}

\section{Exercise 5: DO/DOES or HAVE/HAS? — Choose the Correct Auxiliary}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{\textcolor{correct}{Do}} you have any brothers or sisters?\\
    \textit{(Also correct: Have you got...?)}
    
    \item \textbf{\textcolor{correct}{Has}} she got a new job?\\
    \textit{(Also correct: Does she have...?)}
    
    \item \textbf{\textcolor{correct}{Do}} they have breakfast together every morning?\\
    \textit{(Routine/habit → DO; this is ``have breakfast'' = action)}
    
    \item \textbf{\textcolor{correct}{Have}} you got time to help me?\\
    \textit{(Also correct: Do you have...?)}
    
    \item How often \textbf{\textcolor{correct}{do}} you have English classes?\\
    \textit{(``How often'' = routine → DO required)}
    
    \item \textbf{\textcolor{correct}{Has}} your house got a garden?\\
    \textit{(Also correct: Does your house have...?)}
    
    \item What time \textbf{\textcolor{correct}{does}} he usually have lunch?\\
    \textit{(``Usually'' = routine; ``have lunch'' = action → DOES required)}
    
    \item \textbf{\textcolor{correct}{Have}} you ever had problems with your computer?\\
    \textit{(Present perfect for experience up to now)}
    
    \item \textbf{\textcolor{correct}{Does}} Maria have a car, or does she take the bus?\\
    \textit{(Also correct: Has Maria got...?)}
    
    \item \textbf{\textcolor{correct}{Have}} you got your passport with you?\\
    \textit{(Also correct: Do you have...?)}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{notebox}[Auxiliary Verb Logic]
\textbf{Must use DO/DOES:}
\begin{itemize}
    \item Routines with ``how often,'' ``usually,'' ``always'' + have
    \item HAVE for actions (have breakfast, have a shower)
\end{itemize}

\textbf{Can use HAVE/HAS or DO/DOES:}
\begin{itemize}
    \item Present possession (Have you got...? = Do you have...?)
\end{itemize}

\textbf{Must use HAVE/HAS:}
\begin{itemize}
    \item Present perfect (Have you ever had...?)
\end{itemize}
\end{notebox}

\section{Exercise 6: DID or HAD? — Past Tense Distinctions}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item I was very busy yesterday. I \textbf{\textcolor{correct}{didn't}} have time to call you.\\
    \textit{(Simple past — state at a specific past time)}
    
    \item I couldn't contact you because I \textbf{\textcolor{correct}{didn't have}} my phone.\\
    \textit{(Simple past — didn't have phone [cause] → couldn't contact [result])}
    
    \item \textbf{\textcolor{correct}{Did}} you have any problems on your trip?\\
    \textit{(Simple past question — general query about past period)}
    
    \item By noon yesterday, I \textbf{\textcolor{correct}{hadn't}} had time to eat — but later I did.\\
    \textit{(Past perfect — ``by noon'' marks a point before another past reference)}
    
    \item \textbf{\textcolor{correct}{Did}} she have a good time at the party?\\
    \textit{(Simple past — ``have a good time'' = action)}
    
    \item He was hungry because he \textbf{\textcolor{correct}{hadn't}} had breakfast.\\
    \textit{(Past perfect — cause [not eating] before result [was hungry])}
    
    \item We \textbf{\textcolor{correct}{didn't}} have any trouble finding the restaurant.\\
    \textit{(Simple past — fact about the past; ``have trouble'' = action)}
    
    \item I couldn't pay because I \textbf{\textcolor{correct}{hadn't}} brought my wallet.\\
    \textit{(Past perfect — cause [not bringing] before result [couldn't pay])}
    
    \item \textbf{\textcolor{correct}{Did}} they have enough money to buy the house?\\
    \textit{(Simple past question — asking about state at time of purchase)}
    
    \item The meeting was cancelled because she \textbf{\textcolor{correct}{hadn't}} had time to prepare.\\
    \textit{(Past perfect — cause [no time] before result [meeting cancelled])}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{tipbox}[Decision Rule]
\textbf{Ask yourself:} Is there a CAUSE that happened BEFORE a RESULT?
\begin{itemize}
    \item \textbf{Yes} (two past timepoints in sequence) → \textbf{HAD/HADN'T} (past perfect)
    \item \textbf{No} (single past timeframe) → \textbf{DID/DIDN'T} (simple past)
\end{itemize}

Common patterns for past perfect:
\begin{itemize}
    \item ``because I hadn't...'' (cause before result)
    \item ``by [time], I hadn't...'' (reference point)
    \item ``couldn't... because hadn't...'' (inability due to earlier state)
\end{itemize}
\end{tipbox}

\section{Exercise 7: Why Is This Wrong? — Auxiliary Verb Analysis}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textit{How often have you a shower?}\\
    \textbf{Correction:} How often \textbf{\textcolor{correct}{do you have}} a shower?\\
    \textbf{Why wrong?} ``Have a shower'' = action, not possession. Routines with ``how often'' require DO/DOES. ``Have you...?'' without ``got'' is possible for possession but sounds old-fashioned and is NEVER used for actions.
    
    \vspace{0.3cm}
    
    \item \textit{Was your trip OK? Had you had any problems?}\\
    \textbf{Correction:} Was your trip OK? \textbf{\textcolor{correct}{Did you have}} any problems?\\
    \textbf{Why wrong?} This is a general question about the trip (single past period). Past perfect would only be appropriate if asking about problems BEFORE some other past event (e.g., ``Had you had any problems before the trip was cancelled?'').
    
    \vspace{0.3cm}
    
    \item \textit{Yesterday I was very tired. I hadn't had enough sleep.}\\
    \textbf{Correction:} \textbf{\textcolor{correct}{CORRECT}} (\checkmark)\\
    \textbf{Why correct?} Past perfect is appropriate here: lack of sleep (earlier state) CAUSED tiredness (result). The sequence is: hadn't slept enough → was tired.
    
    \vspace{0.3cm}
    
    \item \textit{Have you got breakfast yet?}\\
    \textbf{Correction:} \textbf{\textcolor{correct}{Have you had}} breakfast yet?\\
    \textbf{Why wrong?} In everyday speech, ``Have you got breakfast yet?'' is common and understood. However, textbook grammar teaches: \textbf{have got} = possession/state, \textbf{have/had} = action. Since ``have breakfast'' is an action (something you do), we use ``Have you had...?'' to keep the rule clear.
    
    \vspace{0.3cm}
    
    \item \textit{We didn't got time to finish the project.}\\
    \textbf{Correction:} We didn't \textbf{\textcolor{correct}{have}} time to finish the project.\\
    \textbf{Why wrong?} After \textbf{didn't}, the verb must be in base form: \textbf{have}, not ``got.'' For simple past, we use ``had'' usually without ``got'' — the standard form taught in English Grammar in Use.
    
    \vspace{0.3cm}
    
    \item \textit{How long have you had got this car?}\\
    \textbf{Correction:} How long \textbf{\textcolor{correct}{have you had}} this car?\\
    \textbf{Why wrong?} In present perfect for duration (``how long''), we use ``have had,'' NOT ``have had got.'' The past participle of ``have'' is ``had.''
\end{enumerate}
\end{answerbox}

\section{Exercise 8: Choose the Correct Form}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{\textcolor{correct}{B) I had}} trouble finding the place.\\
    \textit{(``Have trouble'' = action; talking about completed past experience; explains why late)}
    
    \item \textbf{\textcolor{correct}{B) Did you have}} a good time at the concert last night?\\
    \textit{(``Have a good time'' = action; simple past question about specific past event)}
    
    \item \textbf{\textcolor{correct}{A) didn't have}} her license with her.\\
    \textit{(Simple past: didn't have license [cause] → couldn't drive [result])}
    
    \item \textbf{\textcolor{correct}{B) Had you ... had}} sushi before you went to Japan?\\
    \textit{(Past perfect: asking about experience BEFORE another past event)}
    
    \item \textbf{\textcolor{correct}{B) 're having}} a wonderful time.\\
    \textit{(``Have a ... time'' = action, can use continuous; happening now)}
    
    \item \textbf{\textcolor{correct}{A) do you have}} your car serviced?\\
    \textit{(``How often'' = routine; requires DO; ``have something done'' = causative)}
    
    \item \textbf{\textcolor{correct}{A) Did you have}} any experience when you applied?\\
    \textit{(Simple past: asking about state at time of application)}
    
    \item \textbf{\textcolor{correct}{A) 've got}} two children.\\
    \textit{(Possession; continuous ``are having children'' would mean giving birth now!)}
    
    \item \textbf{\textcolor{correct}{B) was having}} a shower.\\
    \textit{(Past continuous: action in progress when phone rang)}
    
    \item \textbf{\textcolor{correct}{A) Had you had}} problems before the software was installed?\\
    \textit{(Past perfect: problems BEFORE the installation [another past event])}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{pitfallbox}[Watch Out!]
\textbf{Item 8 is a classic trap:}
\begin{itemize}
    \item ``They've got two children'' = possession (\checkmark)
    \item ``They're having two children'' = in the process of giving birth! (\textcolor{red}{wrong meaning})
\end{itemize}

\textbf{Items 7 vs 10:}
\begin{itemize}
    \item Item 7: Single past timeframe (when you applied) → simple past
    \item Item 10: Two past points (problems BEFORE installation) → past perfect
\end{itemize}
\end{pitfallbox}

\section{Exercise 9: Complete the Dialogues}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{At a hotel reception:}\\
    Receptionist: ``\textbf{\textcolor{correct}{Do/Have}} you \textbf{\textcolor{correct}{have/got}} a reservation?''\\
    Guest: ``Yes, I \textbf{\textcolor{correct}{have/I've got}} a booking for two nights.''\\
    \textit{(Present possession — both forms acceptable)}
    
    \vspace{0.2cm}
    
    \item \textbf{At a job interview:}\\
    Interviewer: ``\textbf{\textcolor{correct}{Do/Have}} you \textbf{\textcolor{correct}{have/got}} any experience in this field?''\\
    Candidate: ``Yes, I \textbf{\textcolor{correct}{have/I've got}} three years of experience.''\\
    \textit{(Present possession — both forms acceptable)}
    
    \vspace{0.2cm}
    
    \item \textbf{At a restaurant:}\\
    Waiter: ``\textbf{\textcolor{correct}{Did}} you \textbf{\textcolor{correct}{have}} a nice meal?''\\
    Customer: ``Yes, we \textbf{\textcolor{correct}{had/have had}} a wonderful time. Thank you!''\\
    \textit{(``Have a meal/time'' = action; uses DID in questions)}
    
    \vspace{0.2cm}
    
    \item \textbf{Between friends:}\\
    A: ``Why couldn't you come to the party last night?''\\
    B: ``I \textbf{\textcolor{correct}{didn't have}} time. I \textbf{\textcolor{correct}{had}} too much work to finish.''\\
    \textit{(Past state; simple past preferred in EGU approach)}
    
    \vspace{0.2cm}
    
    \item \textbf{Doctor's appointment:}\\
    Doctor: ``How long \textbf{\textcolor{correct}{have}} you \textbf{\textcolor{correct}{had}} this headache?''\\
    Patient: ``I \textbf{\textcolor{correct}{'ve had}} it for about three days now.''\\
    \textit{(Present perfect for duration — ``how long have you had...'')}
    
    \vspace{0.2cm}
    
    \item \textbf{After an exam:}\\
    A: ``\textbf{\textcolor{correct}{Did}} you \textbf{\textcolor{correct}{have}} any difficulty with the test?''\\
    B: ``Yes, I \textbf{\textcolor{correct}{had}} trouble with the last question.''\\
    \textit{(``Have difficulty/trouble'' = action; uses DID in questions)}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{notebox}[Context Determines Form]
\begin{itemize}
    \item \textbf{Present possession} (items 1, 2): Either form works
    \item \textbf{Actions} (items 3, 6): Must use DO/DID forms
    \item \textbf{Duration with ``how long''} (item 5): Present perfect ``have had''
    \item \textbf{Past inability} (item 4): Can use either simple past or past perfect
\end{itemize}
\end{notebox}

\section{Exercise 10: Error Correction — Find and Fix}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textit{I didn't have breakfast this morning, so I'm hungry now.}\\
    \textbf{\textcolor{correct}{CORRECT}} (\checkmark) — Simple past for completed action; ``have breakfast'' = action.
    
    \item \textit{Have you a moment? I need to ask you something.}\\
    \textbf{Acceptable but formal/old-fashioned.} More natural: ``\textbf{\textcolor{correct}{Do you have}} a moment?'' or ``\textbf{\textcolor{correct}{Have you got}} a moment?''
    
    \item \textit{We didn't have enough money, so we couldn't buy tickets.}\\
    \textbf{\textcolor{correct}{CORRECT}} (\checkmark) — Simple past with cause-result relationship.
    
    \item \textit{Did you have got any problems with your visa application?}\\
    \textbf{Error:} Cannot combine ``did'' with ``got.''\\
    \textbf{Correction:} ``\textbf{\textcolor{correct}{Did you have}} any problems...'' OR ``\textbf{\textcolor{correct}{Had you got}} any problems...''
    
    \item \textit{She has got long hair when she was a child.}\\
    \textbf{Error:} ``When she was a child'' = past; cannot use present ``has got.''\\
    \textbf{Correction:} ``She \textbf{\textcolor{correct}{had}} long hair when she was a child.''
    
    \item \textit{We're having a wonderful holiday — the weather is perfect!}\\
    \textbf{\textcolor{correct}{CORRECT}} (\checkmark) — ``Have a holiday/time'' = action; continuous OK.
    
    \item \textit{How often are you having meetings with your team?}\\
    \textbf{Error:} ``How often'' = routine; routines use simple, not continuous.\\
    \textbf{Correction:} ``How often \textbf{\textcolor{correct}{do you have}} meetings with your team?''
    
    \item \textit{I couldn't enter because I hadn't my key.}\\
    \textbf{Error:} ``Hadn't'' alone (without ``got'') is incomplete for possession.\\
    \textbf{Correction:} ``...because I \textbf{didn't have} my key.''
    
    \item \textit{Have you ever had an accident?}\\
    \textbf{CORRECT} (\checkmark) — Present perfect for experience; ``have an accident'' = action.
    
    \item \textit{Yesterday was terrible. I hadn't had time for anything.}\\
    \textbf{Likely error:} No clear sequence requiring past perfect.\\
    \textbf{Better:} ``Yesterday was terrible. I \textbf{didn't have} time for anything.''\\
    \textit{(However, ``hadn't had'' could be correct if emphasizing cause-result)}
\end{enumerate}
\end{answerbox}

\section{Exercise 11: Tense Sequence Analysis}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{\textcolor{correct}{A:}} \textit{I didn't have time yesterday.}\\
    \textbf{\textcolor{correct}{B:}} \textit{I hadn't had time, so I couldn't finish the report.}
    
    \textbf{Explanation:}
    \begin{itemize}
        \item \textbf{Sentence A (simple past):} States a fact about a single past timeframe. ``Yesterday'' = one point in time. No sequence needed.
        \item \textbf{Sentence B (past perfect):} Shows TWO points in the past with a cause-result relationship. The lack of time (earlier) CAUSED the inability to finish (later).
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{\textcolor{correct}{A:}} \textit{Did you have any problems on the trip?}\\
    \textbf{\textcolor{correct}{B:}} \textit{Had you had any problems before the trip was cancelled?}
    
    \textbf{Explanation:}
    \begin{itemize}
        \item \textbf{Sentence A (simple past):} General question about the trip as one completed experience. Asking about the whole trip period.
        \item \textbf{Sentence B (past perfect):} Asks about problems BEFORE another past event (the cancellation). Two past timepoints in sequence: problems → cancellation.
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{\textcolor{correct}{A:}} \textit{She didn't have her passport.}\\
    \textbf{\textcolor{correct}{B:}} \textit{She couldn't board the plane because she didn't have her passport.}
    
    \textbf{Explanation:}
    \begin{itemize}
        \item \textbf{Sentence A (simple past):} Simple statement of fact. No consequence mentioned, no sequence shown.
        \item \textbf{Sentence B (past perfect):} The ``because'' clause shows cause-result. Not having passport (earlier state) → couldn't board (later result). Past perfect emphasizes the sequence.
    \end{itemize}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{tipbox}[Key Insight]
\textbf{Simple past (DID/DIDN'T):} Reports facts within a single past timeframe.

\textbf{Past perfect (HAD/HADN'T):} Creates a sequence showing:
\begin{enumerate}
    \item An earlier past state/action
    \item A later past result/action
    \item A cause-effect or time-sequence relationship between them
\end{enumerate}

If you can identify TWO past points where one CAUSED or PRECEDED the other, past perfect is appropriate.
\end{tipbox}

\section{Exercise 14: Cambridge FCE — Key Word Transformations}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item She owns three cars. (GOT)\\
    She \textbf{\textcolor{correct}{has got / 's got}} three cars.
    
    \item I couldn't call you because I left my phone at home. (HADN'T)\\
    I couldn't call you because I \textbf{\textcolor{correct}{hadn't got my phone}} with me.\\
    \textit{(Past perfect showing cause before result)}\\
    \textit{Note: The constraint ``HADN'T'' requires past perfect; ``hadn't got'' = hadn't brought.}
    
    \item When did you start working here? (LONG)\\
    How \textbf{\textcolor{correct}{long have you had}} this job?\\
    \textit{(Present perfect with ``how long'' for duration)}
    
    \item Is there a garden in your house? (GOT)\\
    Has \textbf{\textcolor{correct}{your house got a}} garden?
    
    \item We enjoyed ourselves at the party. (TIME)\\
    We \textbf{had a good/great time} at the party.\\
    \textit{(``Have a good time'' = enjoy oneself)}
    
    \item Did you experience any difficulties during the project? (TROUBLE)\\
    Did \textbf{you have any trouble} during the project?\\
    \textit{(``Have trouble'' = experience difficulties)}
\end{enumerate}
\end{answerbox}

\vspace{0.3cm}

\begin{tipbox}[FCE Key Word Transformation Tips]
\begin{itemize}
    \item You must use the key word \textbf{without changing it}
    \item Use between \textbf{2-5 words} including the key word
    \item The meaning must be \textbf{the same} as the original
    \item Watch for tense changes and auxiliary verb requirements
\end{itemize}
\end{tipbox}

\begin{pitfallbox}[Common Errors to Avoid]
\begin{enumerate}
    \item \textcolor{red}{\st{How often have you a shower?}} → \textbf{How often do you have a shower?}
    
    \item \textcolor{red}{\st{Have you got breakfast yet?}} → \textbf{Have you had breakfast yet?}
    
    \item \textcolor{red}{\st{She had got long hair when she was young.}} → \textbf{She had long hair...}
    
    \item \textcolor{red}{\st{Did you have got any problems?}} → \textbf{Did you have any problems?}
    
    \item \textcolor{red}{\st{How long have you had got this car?}} → \textbf{How long have you had this car?}
    
    \item \textcolor{red}{\st{They're having three children.}} → \textbf{They've got three children.}
\end{enumerate}
\end{pitfallbox}
