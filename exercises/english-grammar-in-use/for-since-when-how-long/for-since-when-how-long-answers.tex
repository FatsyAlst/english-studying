\section{Exercise 1: Choose FOR or SINCE}

\begin{enumerate}[leftmargin=2cm]
    \item I've been waiting \textbf{\color{green!70!black}for} over an hour.
    \item She has lived in London \textbf{\color{green!70!black}since} 2015.
    \item They haven't seen each other \textbf{\color{green!70!black}since} last Christmas.
    \item We've known each other \textbf{\color{green!70!black}for} a very long time.
    \item He has been working here \textbf{\color{green!70!black}since} he graduated from university.
    \item I haven't eaten anything \textbf{\color{green!70!black}since} breakfast this morning.
    \item She's been studying English \textbf{\color{green!70!black}for} three years.
    \item They have been married \textbf{\color{green!70!black}since} 1998.
    \item I haven't spoken to him \textbf{\color{green!70!black}since} last Monday.
    \item We've had this car \textbf{\color{green!70!black}for} six months.
    \item He hasn't visited his parents \textbf{\color{green!70!black}since} Easter.
    \item I've been interested in photography \textbf{\color{green!70!black}since} I was a child.
    \item They've been playing tennis \textbf{\color{green!70!black}for} two hours.
    \item She hasn't called me \textbf{\color{green!70!black}since} her birthday.
    \item We've lived in this house \textbf{\color{green!70!black}for} ages.
\end{enumerate}

\vspace{0.5cm}
\textbf{\color{blue!70!black}Rule Reminder:}
\begin{itemize}
    \item Use \textbf{FOR} with a period of time (for 3 years, for 2 hours, for ages)
    \item Use \textbf{SINCE} with a point in time (since 2020, since Monday, since I was a child)
\end{itemize}

\section{Exercise 3: WHEN...? or HOW LONG...?}

\begin{enumerate}[leftmargin=2cm]
    \item \textbf{\color{green!70!black}When} did you start learning English?
    \item \textbf{\color{green!70!black}How long} have you been learning English?
    \item \textbf{\color{green!70!black}When} did they get married?
    \item \textbf{\color{green!70!black}How long} have they been married?
    \item \textbf{\color{green!70!black}When} did she arrive in Brazil?
    \item \textbf{\color{green!70!black}How long} has she been in Brazil?
    \item \textbf{\color{green!70!black}When} did you last see Tom?
    \item \textbf{\color{green!70!black}How long} is it since you last saw Tom?
    \item \textbf{\color{green!70!black}When} did the meeting begin?
    \item \textbf{\color{green!70!black}How long} has the meeting been going on?
    \item \textbf{\color{green!70!black}When} did he move to this city?
    \item \textbf{\color{green!70!black}How long} has he lived in this city?
\end{enumerate}

\vspace{0.5cm}
\textbf{\color{blue!70!black}Rule Reminder:}
\begin{itemize}
    \item Use \textbf{WHEN...?} to ask about a specific time in the past (past simple)
    \item Use \textbf{HOW LONG...?} to ask about a period of time (present perfect)
\end{itemize}

    They bought their house last summer.
    
    \item \textbf{How long have they had their house?} \\
    They have had their house for about six months.
    
    \item \textbf{When did you last see him?} \\
    I last saw him two weeks ago.
    
    \item \textbf{How long is it since you (last) saw him? / How long haven't you seen him?} \\

    The concert started at 8 PM.
    
    \item \textbf{How long has the concert been going on?} \\
    The concert has been going on for two hours.
\end{enumerate}

\section{Exercise 5: Correct the Mistakes}

\begin{enumerate}[leftmargin=2cm]
    \item \sout{I've been living here since three years.} \\
    \textit{Correction:} \textbf{I've been living here for three years.} \\
    \textit{Explanation:} ``Three years'' is a period of time, so use ``for''.
    
    \item \sout{She has worked in that office for 2018.} \\
    \textit{Correction:} \textbf{She has worked in that office since 2018.} \\
    \textit{Explanation:} ``2018'' is a point in time, so use ``since''.
    
    \item \sout{How long did you live in Japan?} (He still lives there) \\
    \textit{Correction:} \textbf{How long have you lived / been living in Japan?} \\
    \textit{Explanation:} Use present perfect because he still lives there. Both simple and continuous forms are correct.
    
    \item \sout{When have you been studying English?} \\
    \textit{Correction:} \textbf{How long have you been studying English?} \\
    \textit{Explanation:} Use ``How long'' for duration, ``When'' for specific time.
    
    \item \sout{I haven't seen her since two weeks.} \\
    \textit{Correction:} \textbf{I haven't seen her for two weeks.} \\
    \textit{Explanation:} ``Two weeks'' is a period of time, so use ``for''.
    
    \item \sout{They are married since 2010.} \\
    \textit{Correction:} \textbf{They have been married since 2010.} \\
    \textit{Explanation:} Use present perfect with ``since'', not simple present.
    
    \item \sout{How long did she start working here?} \\
    \textit{Correction:} \textbf{When did she start working here?} \\
    \textit{Explanation:} Use ``When'' to ask about the starting point, not ``How long''.
    
    \item \sout{I know him since we were at school together.} \\
    \textit{Correction:} \textbf{I have known him since we were at school together.} \\
    \textit{Explanation:} Use present perfect with ``since'', not simple present.
    
    \item \sout{When did you have that car?} (You still have it) \\
    \textit{Correction:} \textbf{How long have you had that car?} \\
    \textit{Explanation:} Use present perfect and ``How long'' because you still have it.
    
    \item \sout{He lives in London for five years.} \\
    \textit{Correction:} \textbf{He has lived / been living in London for five years.} \\
    \textit{Explanation:} Use present perfect with ``for'' + period of time. Both simple and continuous forms are correct.
\end{enumerate}

    \vspace{0.3cm}
    {\Large\bfseries Key Grammar Points Summary}
    \vspace{0.3cm}
\end{center}

\subsection*{FOR vs SINCE}
\begin{itemize}
    \item \textbf{FOR} + period of time (for 3 hours, for 2 weeks, for ages, for a long time)
    \item \textbf{SINCE} + point in time (since Monday, since 2020, since last year, since I was young)
\end{itemize}

\subsection*{WHEN...? vs HOW LONG...?}
\begin{itemize}
    \item \textbf{WHEN...?} asks about a specific time → use Past Simple \\
    \textit{When did you start? / When did it happen?}
    \item \textbf{HOW LONG...?} asks about duration → use Present Perfect \\
    \textit{How long have you been here? / How long has it been?}
\end{itemize}

\subsection*{Verb Tenses}
\begin{itemize}
    \item \textbf{Present Perfect} (have/has + past participle) is used with FOR and SINCE when the action continues to the present or has a connection to the present
    \item \textbf{Past Simple} is used with WHEN to talk about a finished time in the past
\end{itemize}
