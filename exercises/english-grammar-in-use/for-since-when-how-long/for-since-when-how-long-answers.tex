\section{Exercise 1: Choose FOR or SINCE (15 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item I've been waiting \textbf{\textcolor{correct}{for}} over an hour.
    \item She has lived in London \textbf{\textcolor{correct}{since}} 2015.
    \item They haven't seen each other \textbf{\textcolor{correct}{since}} last Christmas.
    \item We've known each other \textbf{\textcolor{correct}{for}} a very long time.
    \item He has been working here \textbf{\textcolor{correct}{since}} he graduated from university.
    \item I haven't eaten anything \textbf{\textcolor{correct}{since}} breakfast this morning.
    \item She's been studying English \textbf{\textcolor{correct}{for}} three years.
    \item They have been married \textbf{\textcolor{correct}{since}} 1998.
    \item I haven't spoken to him \textbf{\textcolor{correct}{since}} last Monday.
    \item We've had this car \textbf{\textcolor{correct}{for}} six months.
    \item He hasn't visited his parents \textbf{\textcolor{correct}{since}} Easter.
    \item I've been interested in photography \textbf{\textcolor{correct}{since}} I was a child.
    \item They've been playing tennis \textbf{\textcolor{correct}{for}} two hours.
    \item She hasn't called me \textbf{\textcolor{correct}{since}} her birthday.
    \item We've lived in this house \textbf{\textcolor{correct}{for}} ages.
\end{enumerate}
\end{answerbox}

\begin{notebox}[Rule Reminder]
\begin{itemize}
    \item Use \textbf{FOR} with a period of time (for 3 years, for 2 hours, for ages)
    \item Use \textbf{SINCE} with a point in time (since 2020, since Monday, since I was a child)
\end{itemize}
\end{notebox}

    \textit{Correction:} \textbf{\textcolor{correct}{I've been living here for three years.}} \\
    \textit{Explanation:} ``Three years'' is a period of time, so use ``for''.
    
    \item \sout{She has worked in that office for 2018.} \\
    \textit{Correction:} \textbf{\textcolor{correct}{She has worked in that office since 2018.}} \\
    \textit{Explanation:} ``2018'' is a point in time, so use ``since''.
    
    \item \sout{How long did you live in Japan?} (He still lives there) \\
    \textit{Correction:} \textbf{\textcolor{correct}{How long have you lived / been living in Japan?}} \\
    \textit{Explanation:} Use present perfect because he still lives there.
    
    \item \sout{When have you been studying English?} \\
    \textit{Correction:} \textbf{\textcolor{correct}{How long have you been studying English?}} \\
    \textit{Explanation:} Use ``How long'' for duration, ``When'' for specific time.
    
    \item \sout{I haven't seen her since two weeks.} \\
    \textit{Correction:} \textbf{\textcolor{correct}{I haven't seen her for two weeks.}} \\
    \textit{Explanation:} ``Two weeks'' is a period of time, so use ``for''.
    
    \item \sout{They are married since 2010.} \\
    \textit{Correction:} \textbf{\textcolor{correct}{They have been married since 2010.}} \\
    \textit{Explanation:} Use present perfect with ``since'', not simple present.
    
    \item \sout{How long did she start working here?} \\
    \textit{Correction:} \textbf{\textcolor{correct}{When did she start working here?}} \\
    \textit{Explanation:} Use ``When'' to ask about the starting point.

    \textit{Correction:} \textbf{\textcolor{correct}{I have known him since we were at school together.}} \\
    \textit{Explanation:} Use present perfect with ``since'', not simple present.
    
    \item \sout{When did you have that car?} (You still have it) \\

    \textit{Explanation:} Use present perfect and ``How long'' because you still have it.

    \textit{Correction:} \textbf{\textcolor{correct}{He has lived / been living in London for five years.}} \\

{\Large\bfseries Key Grammar Points Summary}
\vspace{0.3cm}
\end{center}

\subsection*{FOR vs SINCE}
\begin{itemize}
    \item \textbf{FOR} + period of time (for 3 hours, for 2 weeks, for ages, for a long time)
    \item \textbf{SINCE} + point in time (since Monday, since 2020, since last year, since I was young)
\end{itemize}

\subsection*{WHEN...? vs HOW LONG...?}
\begin{itemize}
    \item \textbf{WHEN...?} asks about a specific time → use Past Simple \\
    \textit{When did you start? / When did it happen?}
    \item \textbf{HOW LONG...?} asks about duration → use Present Perfect \\
    \textit{How long have you been here? / How long has it been?}
\end{itemize}

\subsection*{Verb Tenses}
\begin{itemize}
    \item \textbf{Present Perfect} (have/has + past participle) is used with FOR and SINCE when the action continues to the present
    \item \textbf{Past Simple} is used with WHEN to talk about a finished time in the past
\end{itemize}
