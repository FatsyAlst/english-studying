\begin{grammarbox}[Grammar Summary: FOR and SINCE]
\textbf{FOR} = a \textbf{period of time} (duration)

\vspace{0.2cm}

\begin{tabular}{ll}
\textit{for} & two hours, three days, six months, a long time, ages \\
\end{tabular}

\vspace{0.3cm}

\textbf{SINCE} = a \textbf{point in time} (when something started)

\vspace{0.2cm}

\begin{tabular}{ll}
\textit{since} & Monday, 2015, last summer, I was a child, breakfast \\
\end{tabular}

\vspace{0.3cm}

\textbf{Main Uses:}
\begin{enumerate}[leftmargin=1.5cm]
    \item With \textbf{Present Perfect} to talk about actions from past to present:\\
    \textit{``I have lived here for five years.'' / ``I have lived here since 2019.''}
    
    \item With \textbf{Present Perfect Continuous} for ongoing activities:\\
    \textit{``She has been waiting for an hour.'' / ``She has been waiting since 3 o'clock.''}
\end{enumerate}
\end{grammarbox}

\vspace{0.3cm}

\begin{tipbox}[Tip: WHEN vs. HOW LONG]
\begin{itemize}
    \item \textbf{WHEN...?} asks about a \textbf{specific point in time} → use \textbf{Past Simple}\\
    \textit{``When did you arrive?'' — ``I arrived yesterday.''}
    
    \item \textbf{HOW LONG...?} asks about \textbf{duration} → use \textbf{Present Perfect}\\
    \textit{``How long have you been here?'' — ``I've been here for two hours.''}
\end{itemize}
\end{tipbox}

\begin{tipbox}[Tip: Simple vs. Continuous with FOR/SINCE]
\begin{itemize}
    \item \textbf{Present Perfect Continuous} = emphasizes the \textbf{duration/ongoing process}\\
    \textit{``I've been reading for 3 hours.''} (still reading, focus on duration)
    
    \item \textbf{Present Perfect Simple} = emphasizes the \textbf{result/completion}\\
    \textit{``I've read 3 books.''} (focus on quantity completed)
\end{itemize}
\end{tipbox}
