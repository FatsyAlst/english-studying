\section*{Instructions}

This practice set focuses on two key grammar topics:
\begin{itemize}
    \item Using \textbf{FOR} and \textbf{SINCE} with the present perfect
    \item Using \textbf{WHEN...?} and \textbf{HOW LONG...?} questions
\end{itemize}

Complete all exercises and check your answers with the answer key.

    \item \_\_\_\_\_\_\_\_\_\_ did you start learning English?

\section{Exercise 6: Mixed Practice}

\textit{Complete the conversations with the correct form of the verb and \textbf{for} or \textbf{since}.}

\subsection*{Conversation 1}
\begin{itemize}[label={}, leftmargin=1cm]
    \item \textbf{A:} How long \_\_\_\_\_\_\_\_\_\_ (you / live) in this neighborhood?
    \item \textbf{B:} I \_\_\_\_\_\_\_\_\_\_ (live) here \_\_\_\_\_\_\_ about five years.
    \item \textbf{A:} Do you like it here?
    \item \textbf{B:} Yes, I do. I \_\_\_\_\_\_\_\_\_\_ (be) very happy here \_\_\_\_\_\_\_ I moved in.
\end{itemize}

\subsection*{Conversation 2}
\begin{itemize}[label={}, leftmargin=1cm]
    \item \textbf{A:} \_\_\_\_\_\_\_\_\_\_ (you / know) Sarah long?
    \item \textbf{B:} Yes, we \_\_\_\_\_\_\_\_\_\_ (be) friends \_\_\_\_\_\_\_ we were children.
    \item \textbf{A:} That's nice. When \_\_\_\_\_\_\_\_\_\_ (you / meet) her?
    \item \textbf{B:} We \_\_\_\_\_\_\_\_\_\_ (meet) in elementary school in 2005.
\end{itemize}

\subsection*{Conversation 3}
\begin{itemize}[label={}, leftmargin=1cm]
    \item \textbf{A:} How long \_\_\_\_\_\_\_\_\_\_ (John / work) at the hospital?
    \item \textbf{B:} He \_\_\_\_\_\_\_\_\_\_ (work) there \_\_\_\_\_\_\_ about ten years.
    \item \textbf{A:} When \_\_\_\_\_\_\_\_\_\_ (he / start)?
    \item \textbf{B:} I think he \_\_\_\_\_\_\_\_\_\_ (start) in 2015.
\end{itemize}

\subsection*{Conversation 4}
\begin{itemize}[label={}, leftmargin=1cm]
    \item \textbf{A:} I \_\_\_\_\_\_\_\_\_\_ (not / see) Mark \_\_\_\_\_\_\_ ages. When \_\_\_\_\_\_\_\_\_\_ (you / last / see) him?
    \item \textbf{B:} I \_\_\_\_\_\_\_\_\_\_ (see) him last week at the gym.
    \item \textbf{A:} How is he?
    \item \textbf{B:} He's great! He \_\_\_\_\_\_\_\_\_\_ (have) a new job \_\_\_\_\_\_\_ last month.
\end{itemize}
