% GRAMMAR SUMMARY BOX

\begin{grammarbox}[Grammar Summary: Past Perfect]
\textbf{Form:} \textit{had} + \textbf{past participle} (had done, had seen, had been, etc.)

\vspace{0.3cm}

\textbf{Use:} To talk about something that happened \textbf{before} another past action or time.

\vspace{0.3cm}

\textbf{Timeline:}
\begin{center}
(First action: Past Perfect) $\rightarrow$ (Second action: Past Simple) $\rightarrow$ NOW
\end{center}

\textbf{Examples:}
\begin{itemize}
    \item When I arrived at the cinema, the film \textbf{had already started}. (The film started first)
    \item She wasn't hungry because she \textbf{had just eaten}. (She ate first, then she wasn't hungry)
    \item I didn't recognize him. He \textbf{had changed} so much. (He changed before I saw him)
\end{itemize}

\textbf{Common Time Markers:}
\textit{already, just, never, before, by the time, after, when, until}
\end{grammarbox}

\begin{tipbox}[Tip: Adverb Position]
With Past Perfect, adverbs like \textbf{already}, \textbf{just}, \textbf{never} go between \textbf{had} and the \textbf{past participle}:
\begin{itemize}
    \item She \textbf{had already left}. {[OK]}
    \item She \textbf{already had left}. {[X]}
    \item I \textbf{had never seen} it before. {[OK]}
\end{itemize}
\end{tipbox}

\section{Exercise 2: Past Simple or Past Perfect? (12 points)}

\textit{Complete the sentences with the correct form: past simple or past perfect.}

\begin{enumerate}[label=\arabic*.]
    \item When I \_\_\_\_\_\_\_\_\_\_ (arrive), everybody \_\_\_\_\_\_\_\_\_\_ (already / start) eating.
    \item She \_\_\_\_\_\_\_\_\_\_ (be) very hungry because she \_\_\_\_\_\_\_\_\_\_ (not / eat) anything all day.
    \item I \_\_\_\_\_\_\_\_\_\_ (not / recognize) him because he \_\_\_\_\_\_\_\_\_\_ (grow) a beard.
    \item After he \_\_\_\_\_\_\_\_\_\_ (finish) his work, he \_\_\_\_\_\_\_\_\_\_ (go) home.
    \item When we \_\_\_\_\_\_\_\_\_\_ (get) to the airport, we realized we \_\_\_\_\_\_\_\_\_\_ (forget) our passports.
    \item She \_\_\_\_\_\_\_\_\_\_ (tell) me she \_\_\_\_\_\_\_\_\_\_ (never / try) sushi before.
    \item I \_\_\_\_\_\_\_\_\_\_ (feel) sick because I \_\_\_\_\_\_\_\_\_\_ (eat) too much at dinner.
    \item By the time the ambulance \_\_\_\_\_\_\_\_\_\_ (arrive), the man \_\_\_\_\_\_\_\_\_\_ (already / die).
    \item He \_\_\_\_\_\_\_\_\_\_ (be) late for work because he \_\_\_\_\_\_\_\_\_\_ (miss) his bus.
    \item When I \_\_\_\_\_\_\_\_\_\_ (turn on) the TV, the game \_\_\_\_\_\_\_\_\_\_ (already / finish).
    \item She \_\_\_\_\_\_\_\_\_\_ (pass) her exam because she \_\_\_\_\_\_\_\_\_\_ (study) very hard.
    \item After they \_\_\_\_\_\_\_\_\_\_ (have) dinner, they \_\_\_\_\_\_\_\_\_\_ (watch) a movie.
\end{enumerate}

\begin{pitfallbox}[Common Pitfall: After/Before]
With \textbf{after} and \textbf{before}, the sequence is already clear, so Past Simple is also acceptable:
\begin{itemize}
    \item After he \textbf{had finished / finished} his work, he went home. (Both correct)
    \item Before she \textbf{left / had left}, she checked the windows. (Both correct)
\end{itemize}
However, Past Perfect emphasizes that one action was complete before the other.
\end{pitfallbox}

\vspace{0.5cm}

\begin{examtipbox}[Exam Tip: Narrative Tenses]
In storytelling, use Past Perfect to show \textbf{background information} or \textbf{earlier events}:
\begin{itemize}
    \item Main narrative: Past Simple
    \item Background/earlier: Past Perfect
\end{itemize}
Example: ``I was tired because I \textbf{had worked} all day.'' (The work happened before feeling tired)
\end{examtipbox}

    \item His eyes were red because he \_\_\_\_\_\_\_\_\_\_ (cry).
    
    \item I \_\_\_\_\_\_\_\_\_\_ (read) the book three times before I finally understood it.
    
    \item They were wet because they \_\_\_\_\_\_\_\_\_\_ (swim) in the rain.
    
    \item She \_\_\_\_\_\_\_\_\_\_ (never / taste) sushi before that restaurant.
    
    \item He was out of breath because he \_\_\_\_\_\_\_\_\_\_ (run).
    
    \item I knew the area well because I \_\_\_\_\_\_\_\_\_\_ (live) there for many years.
    
    \item The ground was muddy because it \_\_\_\_\_\_\_\_\_\_ (rain) all night.
    
    \item By 2020, she \_\_\_\_\_\_\_\_\_\_ (visit) every country in Europe.
\end{enumerate}

\begin{tipbox}[When to Use Each Form]
\textbf{Past Perfect Simple:} Completed actions, quantities, number of times\\
\textit{I \textbf{had finished} five reports by noon.}

\textbf{Past Perfect Continuous:} Duration, ongoing activity, visible result\\
\textit{I was tired because I \textbf{had been working} all day.}
\end{tipbox}

Watch out for these commonly confused forms:
\begin{multicols}{3}
\begin{itemize}

    \item rise → risen
    \item go → gone
    \item see → seen
    \item sleep → slept
    \item escape → escaped
    \item know → knew/known
\end{itemize}
\end{multicols}
\end{pitfallbox}

\textbf{1. Action before another past action:}\\
\textit{When I arrived, the film \textbf{had started}.}

\textbf{2. Explaining past situations:}\\
\textit{I was tired because I \textbf{hadn't slept}.}

\textbf{3. Reported speech:}\\
\textit{She said she \textbf{had seen} the movie.}

\textbf{4. Third conditional:}\\
\textit{If I \textbf{had known}, I would have helped.}
