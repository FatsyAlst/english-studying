    \item When I checked my pocket, I realized I \textbf{\textcolor{correct}{had lost}} my keys.

\end{answerbox}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item When I \textbf{\textcolor{correct}{arrived}}, everybody \textbf{\textcolor{correct}{had already started}} eating.
    \item She \textbf{\textcolor{correct}{was}} very hungry because she \textbf{\textcolor{correct}{hadn't eaten}} anything all day.
    \item I \textbf{\textcolor{correct}{didn't recognize}} him because he \textbf{\textcolor{correct}{had grown}} a beard.
    \item After he \textbf{\textcolor{correct}{had finished / finished}} his work, he \textbf{\textcolor{correct}{went}} home.
    \item When we \textbf{\textcolor{correct}{got}} to the airport, we realized we \textbf{\textcolor{correct}{had forgotten}} our passports.
    \item She \textbf{\textcolor{correct}{told}} me she \textbf{\textcolor{correct}{had never tried}} sushi before.
    \item I \textbf{\textcolor{correct}{felt}} sick because I \textbf{\textcolor{correct}{had eaten}} too much at dinner.
    \item By the time the ambulance \textbf{\textcolor{correct}{arrived}}, the man \textbf{\textcolor{correct}{had already died}}.
    \item He \textbf{\textcolor{correct}{was}} late for work because he \textbf{\textcolor{correct}{had missed}} his bus.
    \item When I \textbf{\textcolor{correct}{turned on}} the TV, the game \textbf{\textcolor{correct}{had already finished}}.
    \item She \textbf{\textcolor{correct}{passed}} her exam because she \textbf{\textcolor{correct}{had studied}} very hard.
    \item After they \textbf{\textcolor{correct}{had had / had}} dinner, they \textbf{\textcolor{correct}{watched}} a movie.
\end{enumerate}
\end{answerbox}

\begin{notebox}[Note on ``after'' and ``before'']
When using \textbf{after} or \textbf{before}, both Past Simple and Past Perfect are acceptable because the time sequence is already clear from the conjunction itself.
\end{notebox}

\section{Exercise 4: Write Sentences Using Past Perfect (6 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{\textcolor{correct}{She moved to Paris after she had studied French for five years.}} \\
    \textit{(or: By the time she moved to Paris, she had studied French for five years.)}
    
    \item \textbf{\textcolor{correct}{We went for a walk after the rain had stopped.}} \\
    \textit{(or: After the rain had stopped, we went for a walk.)}
    
    \item \textbf{\textcolor{correct}{He bought a car after he had saved enough money.}} \\
    \textit{(or: After he had saved enough money, he bought a car.)}

    \textit{(or: By the time more guests arrived, they had eaten all the food.)}

    \textit{(or: After I had read the book, I watched the movie.)}
    
    \item \textbf{\textcolor{correct}{We left the venue after the concert had ended.}} \\
    \textit{(or: After the concert had ended, we left the venue.)}
\end{enumerate}
\end{answerbox}

\section{Exercise 8: Past Perfect vs Past Perfect Continuous (10 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item She was exhausted because she \textbf{\textcolor{correct}{had been working}} for 12 hours straight. \\
    \textit{(Continuous — emphasizes duration of activity)}
    
    \item By the time we arrived, they \textbf{\textcolor{correct}{had already eaten}} all the pizza. \\
    \textit{(Simple — completed action, result)}
    
    \item His eyes were red because he \textbf{\textcolor{correct}{had been crying}}. \\
    \textit{(Continuous — visible result of activity)}
    
    \item I \textbf{\textcolor{correct}{had read}} the book three times before I finally understood it. \\
    \textit{(Simple — completed quantity)}
    
    \item They were wet because they \textbf{\textcolor{correct}{had been swimming}} in the rain. \\
    \textit{(Continuous — visible result of activity)}
    
    \item She \textbf{\textcolor{correct}{had never tasted}} sushi before that restaurant. \\
    \textit{(Simple — never + completed experience)}
    
    \item He was out of breath because he \textbf{\textcolor{correct}{had been running}}. \\
    \textit{(Continuous — visible result of activity)}
    
    \item I knew the area well because I \textbf{\textcolor{correct}{had lived / had been living}} there for many years. \\
    \textit{(Both acceptable with ``live'')}
    
    \item The ground was muddy because it \textbf{\textcolor{correct}{had been raining}} all night. \\
    \textit{(Continuous — duration + visible result)}
    
    \item By 2020, she \textbf{\textcolor{correct}{had visited}} every country in Europe. \\
    \textit{(Simple — completed achievement)}
\end{enumerate}
\end{answerbox}

\section{Exercise 9: Correct the Mistakes (10 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \sout{I had forgot my keys} \\
    \textbf{\textcolor{correct}{I had forgotten my keys}} \\
    \textit{Forgot → Forgotten (past participle)}
    
    \item \sout{she has never been to Japan} \\
    \textbf{\textcolor{correct}{she had never been to Japan}} \\
    \textit{Has → Had (backshift in reported speech)}
    
    \item \sout{he had ate dinner} \\
    \textbf{\textcolor{correct}{he had eaten dinner}} \\
    \textit{Ate → Eaten (past participle)}
    
    \item \sout{I didn't knew} \\
    \textbf{\textcolor{correct}{I didn't know}} \\
    \textit{Knew → Know (base form after did)}
    
    \item \sout{They have finished} \\
    \textbf{\textcolor{correct}{They had finished}} \\
    \textit{Have → Had (past perfect, not present perfect)}
    
    \item \sout{had already rose} \\
    \textbf{\textcolor{correct}{had already risen}} \\
    \textit{Rose → Risen (past participle)}
    
    \item \sout{she didn't slept well} \\
    \textbf{\textcolor{correct}{she hadn't slept well}} \\
    \textit{Didn't sleep → Hadn't slept (past perfect for reason)}
    
    \item \sout{had already escape} \\
    \textbf{\textcolor{correct}{had already escaped}} \\
    \textit{Escape → Escaped (past participle)}
    
    \item \sout{I had went} \\
    \textbf{\textcolor{correct}{I had gone}} \\
    \textit{Went → Gone (past participle)}
    
    \item \sout{he had see} \\
    \textbf{\textcolor{correct}{he had seen}} \\
    \textit{See → Seen (past participle)}
\end{enumerate}
\end{answerbox}

\textbf{Scoring:} 1 point per correct answer. Total: 10 points.

\section{Exercise 12: Third Conditional with Past Perfect (6 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item If I \textbf{\textcolor{correct}{had known}} about the traffic, I would have left earlier.
    
    \item She wouldn't have failed the exam if she \textbf{\textcolor{correct}{had studied}} harder.
    
    \item If they \textbf{\textcolor{correct}{had told}} us the truth, we wouldn't have been so angry.
    
    \item I would have called you if I \textbf{\textcolor{correct}{hadn't lost / had not lost}} your number.
    
    \item If the weather \textbf{\textcolor{correct}{had been}} better, we would have gone to the beach.
    
    \item He wouldn't have missed the flight if he \textbf{\textcolor{correct}{had left}} home earlier.
\end{enumerate}
\end{answerbox}

\begin{notebox}[Third Conditional Structure]
\begin{center}
\textbf{If} + \textbf{Past Perfect}, + \textbf{would have} + past participle
\end{center}
Used for \textbf{hypothetical past situations} — things that didn't actually happen.
\end{notebox}

\section{Exercise 14: Cambridge Use of English (8 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item I was surprised to meet John because I \textbf{\textcolor{correct}{hadn't seen him for}} five years. \\
    \textit{(FOR — with duration "five years"; SINCE would need a point in time like "since 2019")}

    \textit{(ALREADY — between had and past participle)}

    \textit{(WORKED — past participle of work)}
    
    \item When I arrived, Tom \textbf{\textcolor{correct}{had just left}}. \\
    \textit{(JUST — between had and past participle)}
    
    \item When I tried to pay, I realized I \textbf{\textcolor{correct}{had forgotten}} my wallet. \\
    \textit{(FORGOTTEN — past participle of forget)}
    
    \item I \textbf{\textcolor{correct}{had never eaten / tasted}} Thai food before. \\
    \textit{(NEVER — between had and past participle)}
    
    \item He went to the gym \textbf{\textcolor{correct}{after he had finished}} work. \\
    \textit{(AFTER — with past perfect)}
    
    \item When we got there, we realized the restaurant \textbf{\textcolor{correct}{had already closed}}. \\
    \textit{(ALREADY — between had and past participle)}
\end{enumerate}
\end{answerbox}

\section{Exercise 15: TOEFL Integrated Skills (8 points)}

\begin{answerbox}[Answers]
\begin{enumerate}[label=\arabic*.]
    \item \textbf{What did scientists know before Fleming's discovery?}\\
    \textbf{\textcolor{correct}{Before Fleming's discovery, scientists had known that certain molds could kill bacteria, but no one had successfully isolated the active compound.}}
    
    \item \textbf{What happened before Fleming noticed the contaminated culture?}\\
    \textbf{\textcolor{correct}{Before Fleming noticed the contaminated culture, a mold had contaminated one of his bacterial cultures. / Earlier researchers had observed similar phenomena but had not investigated further.}}

    \textbf{\textcolor{correct}{Before Fleming discovered penicillin, no one had isolated the compound.}}\\

\end{enumerate}
\end{answerbox}

\subsection*{Past Perfect Formation}
\begin{center}
\begin{tabular}{|l|l|}
\hline
\textbf{Positive} & had + past participle \\
\hline
\textbf{Negative} & had not (hadn't) + past participle \\
\hline
\textbf{Question} & Had + subject + past participle...? \\
\hline
\end{tabular}
\end{center}

\subsection*{Main Uses}
\begin{itemize}
    \item \textbf{Action before another past action:} \textit{When I arrived, the film had started.}
    \item \textbf{Explaining past situations:} \textit{I was tired because I hadn't slept well.}
    \item \textbf{Reported speech:} \textit{She said she had finished.}
    \item \textbf{Third conditional:} \textit{If I had known, I would have helped.}
\end{itemize}
