\section{Exercise 1: Complete with Past Perfect}

\begin{enumerate}[leftmargin=2cm]
    \item When I arrived at the party, Tom \textbf{\color{green!70!black}had already left}.
    \item She didn't want to go to the cinema because she \textbf{\color{green!70!black}had already seen} the film.
    \item By the time I got to the station, the train \textbf{\color{green!70!black}had already gone}.
    \item I was very tired because I \textbf{\color{green!70!black}hadn't slept / had not slept} well the night before.
    \item They \textbf{\color{green!70!black}had never met} before the conference.
    \item When we arrived at the restaurant, they \textbf{\color{green!70!black}had given} our table to someone else.
    \item I didn't recognize her because she \textbf{\color{green!70!black}had changed} so much.
    \item He couldn't pay for the meal because he \textbf{\color{green!70!black}had forgotten} his wallet at home.
    \item By 10 PM, I \textbf{\color{green!70!black}had finished} all my homework.
    \item She told me she \textbf{\color{green!70!black}had been} to Paris three times before.
    \item The streets were wet because it \textbf{\color{green!70!black}had rained / had been raining} all morning.
    \item I \textbf{\color{green!70!black}had just got / gotten} home when the phone rang.
    \item They \textbf{\color{green!70!black}had known} each other for years before they started dating.
    \item When I checked my pocket, I realized I \textbf{\color{green!70!black}had lost} my keys.
    \item The movie \textbf{\color{green!70!black}had already started} when we entered the cinema.
\end{enumerate}

\vspace{0.5cm}
\textbf{\color{blue!70!black}Rule Reminder:}
\begin{itemize}
    \item Past Perfect = \textbf{had + past participle}
    \item Use it when one past action happened \textbf{before} another past action
    \item Common time expressions: \textit{already, just, never, before, by the time, when}
\end{itemize}

\section{Exercise 2: Past Simple or Past Perfect?}

\begin{enumerate}[leftmargin=2cm]
    \item When I \textbf{arrived}, everybody \textbf{had already started} eating.
    \item She \textbf{was} very hungry because she \textbf{hadn't eaten / had not eaten} anything all day.
    \item I \textbf{didn't recognize} him because he \textbf{had grown} a beard.
    \item After he \textbf{had finished} his work, he \textbf{went} home.
    \item When we \textbf{got} to the airport, we realized we \textbf{had forgotten} our passports.
    \item She \textbf{told} me she \textbf{had never tried} sushi before.
    \item I \textbf{felt} sick because I \textbf{had eaten} too much at dinner.
    \item By the time the ambulance \textbf{arrived}, the man \textbf{had already died}.
    \item He \textbf{was} late for work because he \textbf{had missed} his bus.
    \item When I \textbf{turned on} the TV, the game \textbf{had already finished}.
    \item She \textbf{passed} her exam because she \textbf{had studied} very hard.
    \item After they \textbf{had had} dinner, they \textbf{watched} a movie.
\end{enumerate}

\vspace{0.5cm}
\textbf{\color{blue!70!black}Key Point:}
\begin{itemize}
    \item Use \textbf{Past Simple} for the main event in the story timeline
    \item Use \textbf{Past Perfect} for the action that happened \textbf{before} the main event
    \item After ``after'' + past perfect, use past simple for the next action
\end{itemize}

\section{Exercise 3: Write Sentences Using Past Perfect}

\section{Exercise 5: Correct the Mistakes}

\begin{enumerate}[leftmargin=2cm]
    \item \sout{When I got home, I realized I had forgot my keys at work.} \\
    \textit{Correction:} \textbf{When I got home, I realized I had forgotten my keys at work.}\\
    \textit{Explanation:} The past participle of ``forget'' is ``forgotten'', not ``forgot''.
    
    \item \sout{She told me she has never been to Japan before.} \\
    \textit{Correction:} \textbf{She told me she had never been to Japan before.}\\
    \textit{Explanation:} Use past perfect (had been), not present perfect (has been), in reported speech about the past.
    
    \item \sout{After he had ate dinner, he went to bed.} \\
    \textit{Correction:} \textbf{After he had eaten dinner, he went to bed.}\\
    \textit{Explanation:} The past participle of ``eat'' is ``eaten'', not ``ate'' (which is past simple).
    
    \item \sout{I didn't knew that you had already left.} \\
    \textit{Correction:} \textbf{I didn't know that you had already left.}\\
    \textit{Explanation:} After ``didn't'', use the base form ``know'', not ``knew''.
    
    \item \sout{They have finished all the work before the boss arrived.} \\
    \textit{Correction:} \textbf{They had finished all the work before the boss arrived.}\\
    \textit{Explanation:} Use past perfect (had finished) to show an action before another past action.
    
    \item \sout{By the time I woke up, the sun had already rose.} \\
    \textit{Correction:} \textbf{By the time I woke up, the sun had already risen.}\\
    \textit{Explanation:} The past participle of ``rise'' is ``risen'', not ``rose'' (which is past simple).
    
    \item \sout{She was tired because she didn't slept well.} \\
    \textit{Correction:} \textbf{She was tired because she hadn't slept well.}\\
    \textit{Explanation:} Use past perfect negative (hadn't slept) for the earlier action. Also, after ``didn't'', use base form.
    
    \item \sout{When the police arrived, the thief had already escape.} \\
    \textit{Correction:} \textbf{When the police arrived, the thief had already escaped.}\\
    \textit{Explanation:} The past participle of ``escape'' is ``escaped''.
    
    \item \sout{I had went to that restaurant many times before it closed.} \\
    \textit{Correction:} \textbf{I had been / gone to that restaurant many times before it closed.}\\
    \textit{Explanation:} The past participle of ``go'' is ``gone'' or ``been'' (for places), not ``went''.
    
    \item \sout{He told me he had see the movie twice.} \\
    \textit{Correction:} \textbf{He told me he had seen the movie twice.}\\
    \textit{Explanation:} The past participle of ``see'' is ``seen'', not ``see''.
\end{enumerate}

\section{Exercise 6: Questions with Past Perfect}

\begin{enumerate}[leftmargin=2cm]
    \item \textbf{Had you ever visited} London before your trip last year?
    \item \textbf{Had she finished} her homework before the teacher arrived?
    \item \textbf{Had they already eaten} when you got to the restaurant?
    \item \textbf{Had he met} her before the party?
    \item Why \textbf{hadn't you told / had you not told} me about the problem earlier?
    \item How long \textbf{had they been} married before they had children?
    \item What \textbf{had happened} before the police arrived?
    \item Where \textbf{had she worked} before she joined our company?
\end{enumerate}

\vspace{0.5cm}
\textbf{\color{blue!70!black}Question Formation:}
\begin{itemize}
    \item Yes/No questions: \textbf{Had + subject + past participle...?}
    \item Wh- questions: \textbf{Wh-word + had + subject + past participle...?}
    \item Negative questions: \textbf{Hadn't + subject + past participle...?}
\end{itemize}

\subsection*{Past Perfect Formation}
\begin{itemize}
    \item \textbf{Affirmative:} Subject + had + past participle\\
    \textit{I had finished / She had gone / They had seen}
    \item \textbf{Negative:} Subject + hadn't + past participle\\
    \textit{I hadn't finished / She hadn't gone / They hadn't seen}
    \item \textbf{Question:} Had + subject + past participle?\\
    \textit{Had you finished? / Had she gone? / Had they seen?}
\end{itemize}

\subsection*{When to Use Past Perfect}
\begin{itemize}
    \item To show that one action happened \textbf{before} another action in the past
    \item With time expressions like: \textit{before, after, by the time, when, already, just, never, until}
    \item In reported speech when the original statement was in present perfect or past simple
\end{itemize}

\subsection*{Common Irregular Past Participles}
\begin{itemize}
    \item be → been, do → done, go → gone/been, see → seen
    \item eat → eaten, forget → forgotten, give → given, take → taken
    \item write → written, speak → spoken, break → broken, choose → chosen
    \item rise → risen, fall → fallen, know → known, grow → grown
\end{itemize}
